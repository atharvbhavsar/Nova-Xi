# ============================================================================
# INSTALLATION INSTRUCTIONS
# ============================================================================

# Step 1: Install Node.js v18+ and Git
#   Windows: Download from nodejs.org and git-scm.com
#   Linux:   sudo apt-get install nodejs npm git
#   Mac:     brew install node git

# Step 2: Clone repository
#   git clone https://github.com/yourusername/tokenized-academic-credentials.git
#   cd tokenized-academic-credentials

# Step 3: Install all dependencies (choose one method)
#
#   Method A - Installation Script (Recommended):
#     Windows:    .\install.ps1
#     Linux/Mac:  chmod +x install.sh && ./install.sh
#
#   Method B - npm Command:
#     npm run install-all
#
#   Method C - Manual:
#     npm install
#     cd frontend && npm install && cd ..

# Step 4: Configure environment variables
#   cp .env.example .env
#   cp frontend/.env.example frontend/.env
#   # Edit both .env files with your API keys

# Step 5: Compile and test
#   npm run compile
#   npm run test

# Step 6: Run application
#   Terminal 1: npm run node        (start blockchain)
#   Terminal 2: npm run deploy:local (deploy contract)
#   Terminal 3: npm run frontend     (start web app)
